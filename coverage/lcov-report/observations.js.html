var express = require('express');
var router = express.Router();
 
const dsn = process.env.DBWEBB_DSN || 'mongodb://127.0.0.1:27017/birdreport';
const Obs = require('mongo-crud-simple')(dsn, 'observations');
 
// Obeservations page, all observations
router.get('/', async (req, res, next) => {
    try {
        let data = await Obs.index();
 
        res.render('observations/index', {
            title: 'Observationer',
            observations: data
        });
    } catch (error) {
        next(error);
    }
});
 
router.get('/create', (req, res) => {
    res.render('observations/create', {
        title: 'Ny observation'
    });
});
 
router.post('/create', async (req, res, next) => {
    try {
        let obs = {
            bird: req.body.bird,
            date: req.body.date,
            location: req.body.location,
            text: req.body.text,
            image: req.body.image,
            user: req.body.user
        };
 
        await Obs.create(obs);
    } catch (error) {
        next(error);
    }
 
    res.redirect('/observations');
});
 
module.exports = router;
